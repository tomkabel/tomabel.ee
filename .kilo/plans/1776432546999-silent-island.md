### React Best Practices Application

Based on `react-best-practices.md` (Vercel React Guidelines):

#### Critical Priority - Bundle Size & Waterfalls
| Rule | Application |
|------|-------------|
| `bundle-dynamic-imports` | VideoPlayer loaded dynamically to reduce initial bundle |
| `bundle-barrel-imports` | Import video.js icons directly, not from main module |

#### Medium Priority - Re-render Optimization
| Rule | Application |
|------|-------------|
| `rerender-memo` | Memoize `AnimatedBackground` blobs (static, don't re-render) |
| `rerender-functional-setstate` | Use functional `setVideoEnded(prev => true)` pattern |
| `rerender-defer-reads` | Extract `handleVideoEnded` callback, avoid subscribing to videoEnded state |
| `rerender-transitions` | Wrap completion state update in `startTransition` if needed |

#### Medium Priority - Rendering Performance
| Rule | Application |
|------|-------------|
| `rendering-hoist-jsx` | Extract static `BackLinkArrow` SVG icon outside component |
| `rendering-conditional-render` | Use ternary for `videoEnded` condition, not `&&` |

#### Low-Medium Priority - JS Performance
| Rule | Application |
|------|-------------|
| `js-early-exit` | Return early from error states in VideoPlayer |
| `js-hoist-regexp` | Avoid creating RegExp in render loops |

### Code Patterns to Apply

```tsx
// ✅ GOOD: Functional setState (rerender-functional-setstate)
const handleVideoEnded = useCallback(() => {
  setVideoEnded(prev => true);  // Functional, stable callback
}, []);

// ✅ GOOD: Memoized static background (rerender-memo)
const AnimatedBackground = memo(function AnimatedBackground() {
  return (/* animated blobs */);
});

// ✅ GOOD: Hoist static JSX (rendering-hoist-jsx)
const BackLinkArrow = () => (
  <svg className="..." aria-hidden="true">
    <path d="M10 19l-7-7m0 0l7-7m-7 7h18" />
  </svg>
);

// ✅ GOOD: Ternary for conditionals (rendering-conditional-render)
{videoEnded ? (
  <CompletionCard />
) : null}

// ❌ BAD: Object conditional (not allowed per rendering-conditional-render)
// {videoEnded && <CompletionCard />}  // Avoid && for renders
```

## Phase 4: New Code Structure

### VooglaadijaPresentation.tsx Structure

```tsx
import { useState, useCallback, memo } from 'react';
import VooglaadijaVideoPlayer from './VooglaadijaVideoPlayer';

// ✅ GOOD: Hoisted static SVG (rendering-hoist-jsx)
const BackLinkArrow = () => (
  <svg
    className="w-4 h-4 transform group-hover:-translate-x-1 transition-transform"
    fill="none"
    stroke="currentColor"
    viewBox="0 0 24 24"
    aria-hidden="true"
  >
    <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M10 19l-7-7m0 0l7-7m-7 7h18" />
  </svg>
);

// ✅ GOOD: Memoized background (rerender-memo)
const AnimatedBackground = memo(function AnimatedBackground() {
  return (
    <div className="absolute inset-0 w-full h-full overflow-hidden pointer-events-none" aria-hidden="true">
      <Blob position="top-1/4 left-1/4" color="indigo" />
      <Blob position="bottom-1/4 right-1/4" color="purple" />
      <Blob position="top-1/2 left-1/2" color="indigo-dark" />
    </div>
  );
});

export default function VooglaadijaPresentation() {
  // ✅ GOOD: Functional setState (rerender-functional-setstate)
  const [videoEnded, setVideoEnded] = useState(false);

  // ✅ GOOD: useCallback for stable reference
  const handleVideoEnded = useCallback(() => {
    setVideoEnded(prev => true);
  }, []);

  return (
    <div className="relative min-h-screen w-full overflow-hidden animated-bg">
      <AnimatedBackground />

      <main
        role="main"
        className="relative z-10 min-h-screen flex flex-col items-center justify-center px-4"
      >
        <div className="glass-container">
          <Badge>VOOGLAADIJA</Badge>

          <header className="text-center mb-8">
            <h1 className="font-['Staatliches'] text-4xl md:text-5xl lg:text-6xl text-white tracking-wider">
              Video Esitlus
            </h1>
            <p className="text-slate-300 mt-3 font-medium">
              Vooglaadija 2026
            </p>
          </header>

          <section aria-label="Video player">
            <VideoContainer>
              <VooglaadijaVideoPlayer
                src="/vooglaadija-presentation-1.mp4"
                onEnded={handleVideoEnded}
              />
            </VideoContainer>
          </section>

          {/* ✅ GOOD: Ternary instead of && (rendering-conditional-render) */}
          {videoEnded ? (
            <CompletionCard />
          ) : null}
        </div>

        <footer className="mt-12 pb-8">
          <a
            href="/"
            className="inline-flex items-center gap-2 text-slate-400 hover:text-white transition-colors text-sm group focus-visible:ring-2 focus-visible:ring-indigo-500 focus-visible:outline-none rounded-md px-2 py-1"
            aria-label="Tagasi pealehele"
          >
            <BackLinkArrow />
            Tagasi pealehele
          </a>
        </footer>
      </main>
    </div>
  );
}
```
